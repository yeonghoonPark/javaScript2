location 개체

        location.href = 'https://www.naver.com';
        let hostDomain = location.hostname; // 도메인
        console.log(hostDomain);
        let hostPath = location.pathname; // 경로 파일명
        console.log(hostPath);
        let hostPro = location.protocol; // 접속 프로토콜
        console.log(hostPro);

history 개체

        history object (현재문서에서는 오류남, 주석처리)
        history.back(); // 뒤로가기, 현재 문서에서는 뒤가 없기 때문에 불가능
        history.forward(); // 앞으로가기, 현재 문서에서는 앞이 없기 때문에 불가능
        history.go(number); // number만큼 백하거나 포워드하거나, 양수냐 음수냐에 따라
