수학개체

        let pi = Math.PI;
        let r = 5;
        let width = Math.pow(r,2)*pi;
        console.log(width); 
        console.log(Math.abs(-5)); // 절대값
        console.log(Math.max(-5,5,1,2,66,4,7)); // 최대값
        console.log(Math.min(-5,5,1,2,66,4,7)); // 최소값
        console.log(Math.round(447.2465,2)); // 반올림
        console.log(Math.ceil(447.2465)); // 올림
        console.log(Math.floor(447.2465)); // 내림
        console.log(Math.sqrt(4)); // 루트
        console.log(Math.pow(4, 3)); // 제곱
        console.log(Math.random()); // 무작위
        console.log(Math.PI); // 원주율
